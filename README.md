# WGA with Progressive Cactus

The alignment was accomplished with Progressive Cactus v2.9.3. The approach shown in this repo follows the steps detailed by the software authors: https://github.com/ComparativeGenomicsToolkit/cactus/blob/master/doc/progressive.md

All the steps below intend to split the workload into different jobs to speed up the aligning process. This should not be necessary for newer Cactus versions by using the `--script` flag. 

## cactus-prepare
```bash
# Set certificates path to avoid cactus being killed because of certificate verification failures
export SSL_CERT_FILE=/etc/ssl/certs/ca-certificates.crt

# Set try
set -o nounset -o errexit -o xtrace

#Set directory
cd /lustre/scratch/frcastel/21families/ || exit

# Creating variables
readonly NUM_CORES=$SLURM_NTASKS
readonly CONTAINER=/lustre/work/frcastel/software/
readonly WORKDIR=cactus_matador
readonly SEQFILE=wga_21families.txt
readonly OUTSEQFILE=bats_21families_tree.txt
readonly OUTPUTHAL=wga_21families_bats.hal
readonly CACTUS_OPTIONS="--disableCaching --cleanWorkDir onSuccess --gpu --maxCores $SLURM_NTASKS --lastzCores $SLURM_GPUS_ON_NODE --doubleMem true"
readonly CACTUS_PREPARE_FILE=cactus_prepare_21families.txt
readonly THIS_SCRIPT=wga_21families_matador.sh

# Create directory or move to next command
if [ -d $WORKDIR ]; then
    echo "# workDir already exists"
else
    mkdir -p $WORKDIR
fi

#if cactus_prepare_matador.txt and cactus_matador/0 exist, then move to next command
if [ -f $CACTUS_PREPARE_FILE ] && [ -d $WORKDIR/0 ]; then
    echo "# $CACTUS_PREPARE_FILE and $WORKDIR/0 already exist. Running cactus pipeline"
else
    # Get commands for the cactus pipeline
    singularity run -B $(pwd):$(pwd) $CONTAINER/cactus_v2.9.1-gpu.sif cactus-prepare bin/"$SEQFILE" --outDir $WORKDIR --jobStore $WORKDIR --outSeqFile $WORKDIR/"$OUTSEQFILE" --outHal $WORKDIR/"${OUTPUTHAL}" --cactusOptions "$CACTUS_OPTIONS" > bin/"$CACTUS_PREPARE_FILE"

    # wait until cactus_prepare_matador.txt is created and then move to next command
    while [ ! -f bin/"$CACTUS_PREPARE_FILE" ]; do
        sleep 1
    done

    # Append commands generated by cactus-prepare to the script
    # First add a line in the end of the cactus_matador script
    echo >> "$THIS_SCRIPT"
    cat bin/"$CACTUS_PREPARE_FILE" >> bin/"$THIS_SCRIPT"

fi

```
